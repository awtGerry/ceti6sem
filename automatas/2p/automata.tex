\documentclass[border=0.9cm]{standalone}

% required packages and libraries
\usepackage{tikz}
\usetikzlibrary{automata, arrows.meta, positioning}

\begin{document}

\begin{tikzpicture} [node distance = 3cm, on grid, auto]

\node (a) [state, initial, initial text = {}] {$a$};
\node (c) [state, above right = of a] {$c$};
\node (b) [state, below right = of a] {$b$};
\node (d) [state, right = of b] {$d$};
\node (e) [state, accepting, right = of c] {$e$};

\path [-stealth, thick]
	(a) edge node {$b$}   (c)
	(a) edge node {$a$}   (b)
	(c) edge node {$a$}   (b)
	(c) edge [loop above]  node {$b$}()
	(b) edge [loop below]  node {$b$}()
	(b) edge[bend left] node {$b$} (d)
    (d) edge[bend left] node {$a$} (b)
    (e) edge node {$b$} (c)
    (e) edge node {$a$} (b)
    (d) edge node {$b$} (e);
\end{tikzpicture}

\end{document}
